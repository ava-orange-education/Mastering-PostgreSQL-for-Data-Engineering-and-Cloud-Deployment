
CREATE TYPE day_of_week AS ENUM ('Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday');


CREATE TABLE schedule (
    id SERIAL PRIMARY KEY,
    event_name VARCHAR(100),
    event_day day_of_week
);
