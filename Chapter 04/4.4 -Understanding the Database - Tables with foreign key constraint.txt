CREATE TABLE parent (
    id SERIAL PRIMARY KEY,
    name VARCHAR(50)
);


CREATE TABLE child (
    id SERIAL PRIMARY KEY,
    parent_id INTEGER REFERENCES parent(id) ON DELETE CASCADE ON UPDATE CASCADE,
    value VARCHAR(50)
);
